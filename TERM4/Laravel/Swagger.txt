

<<===== Step in the install Swagger ======>>



Step 1: Install L5-Swagger
composer require darkaonline/l5-swagger

Step 2: Publish the Configuration
php artisan vendor:publish --provider="L5Swagger\L5SwaggerServiceProvider"

Step 3: Configure L5-Swagger
Open the config/l5-swagger.php file

Step 4: Add Swagger Annotations to Your Controllers
/**
 * @OA\Get(
 *     path="/api/users",
 *     summary="Get list of users",
 *     tags={"Users"},
 *     @OA\Response(
 *         response=200,
 *         description="Successful operation"
 *     ),
 *     @OA\Response(
 *         response=401,
 *         description="Unauthenticated"
 *     ),
 *     @OA\Response(
 *         response=403,
 *         description="Forbidden"
 *     )
 * )
 */
public function index()
{
    // Your code here
}

Step 5: Generate Swagger Documentation
php artisan l5-swagger:generate

Step 6: Access Swagger UI
