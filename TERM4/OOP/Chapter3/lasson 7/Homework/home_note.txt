
export class Shape{
    constructor(protected leftX: number, protected bottomY:number){}
    getWidth():number{
        return this.leftX - this.leftX;
    }
    getHeight():number{
        return this.bottomY - this.bottomY;
    }
    getArea():number{
        return this.getWidth() * this.getHeight();
    }
}
